    0x01解决时区问题

    添加tzdata可以解决时间问题

    Dockfile:

    FROM postgres:12.3-alpine

RUN sed -i 's/dl-cdn.alpinelinux.org/mirrors.aliyun.com/g' /etc/apk/repositories &&\
apk add --no-cache tzdata &&\
cp /usr/share/zoneinfo/Asia/Shanghai /etc/localtime
